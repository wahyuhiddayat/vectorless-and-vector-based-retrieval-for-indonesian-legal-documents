%% KEY: all horizontal spacing via positioning library with FIXED gaps.
%%  \intra = gap between boxes inside a stage
%%  \inter = gap between boxes across stages (slightly larger so region gap is uniform)
\def\intra{1.5cm}
\def\inter{2.4cm}

\begin{document}
\begin{tikzpicture}[node distance=\inter]

%% ── Spine, left to right, relative positioning ──────────────────────────────
\node[query] (query) {Query $q$};
\node[proc, right=\intra of query] (docsel) {BM25 over docs};

\node[docbox, right=\inter of docsel] (doc) {Document $D_i$};

\node[tnode, right=\inter of doc, yshift=2.1cm] (n1) {Internal node};
\node[tnode, right=\inter of doc, yshift=-2.1cm] (n2) {Internal node};

%% leaves: place relative to nodes, fixed vertical spacing
\node[leaf, right=\inter of n1, yshift=1.05cm] (l1) {$\ell_1$};
\node[leaf, right=\inter of n1, yshift=-1.05cm] (l2) {$\ell_2$};
\node[leaf, right=\inter of n2, yshift=1.05cm] (l3) {$\ell_3$};
\node[leaf, right=\inter of n2, yshift=-1.05cm] (l4) {$\ell_4$};

%% Pool box drawn on background; leaves above it on main layer stay visible.
%% Region is also background but declared later, so draw pool here on background
%% BEFORE region by using a dedicated earlier background pass is complex;
%% instead keep pool on MAIN layer but redraw leaves on top of it.
\node[pooldash, fit=(l1)(l2)(l3)(l4), inner sep=7pt] (pool) {};
\foreach \n/\t in {l1/\ell_1, l2/\ell_2, l3/\ell_3, l4/\ell_4} {
  \node[leaf] at (\n) {$\t$};
}

%% Re-ranking: continue spine from pool at the doc's y-level
\node[proc, right=\inter of pool] (merge) {Merge candidates};
\node[proc, right=\intra of merge] (rerank) {Global re-rank};
\node[result, right=\intra of rerank] (sources) {Ranked\\sources};

%% ── Notes (italic, outside) ─────────────────────────────────────────────────
\node[note, below=0.5cm of docsel] (docselnote) {score by metadata, summaries\\keep top-$K_{\text{doc}}$};
\node[note, below=0.5cm of doc] (docnote) {for each retrieved $D_i$};
\node[note] (nnote) at ($(n2)+(0,-1.3)$) {beam scored by\\title, summary};
\node[note, below=0.4cm of pool] (poolnote) {candidate leaf pool,\\re-ranked per $D_i$\\by full-text BM25};
\node[note, below=0.5cm of merge] (mergenote) {dedup by doc, node};
\node[note, below=0.5cm of rerank] (reranknote) {score by full text\\keep top-$K$};

%% ── Regions (background, fit content) ───────────────────────────────────────
\begin{scope}[on background layer]
  \node[region, fit=(query)(docsel)(docselnote), inner xsep=18pt, inner ysep=18pt] (reg1) {};
  \node[region, fit=(doc)(docnote)(n1)(n2)(nnote)(pool)(poolnote),
        inner xsep=18pt, inner ysep=18pt] (reg2) {};
  \node[region, fit=(merge)(mergenote)(rerank)(reranknote)(sources),
        inner xsep=18pt, inner ysep=18pt] (reg3) {};
\end{scope}
\node[regtitle, anchor=south] at ($(reg1.north)+(0,0.18)$) {Document retrieval};
\node[regtitle, anchor=south] at ($(reg2.north)+(0,0.18)$) {Tree traversal};
\node[regtitle, anchor=south] at ($(reg3.north)+(0,0.18)$) {Re-ranking};

%% ── Flow ────────────────────────────────────────────────────────────────────
\draw[flow]   (query)  -- (docsel);
\draw[flow]   (docsel.east) -- (doc.west);

\draw[branch] (doc.east) -- (n1.west);
\draw[branch] (doc.east) -- (n2.west);
\draw[branch] (n1.east) -- (l1.west);
\draw[branch] (n1.east) -- (l2.west);
\draw[branch] (n2.east) -- (l3.west);
\draw[branch] (n2.east) -- (l4.west);

\draw[flow] (pool.east) -- (merge.west);
\draw[flow] (merge)  -- (rerank);
\draw[flow] (rerank) -- (sources);

\end{tikzpicture}
\end{document}
